            if actual == expected:
                print(f"PASS: line_{i} - {actual} words")
            else:
                print(f"FAIL: line_{i} - expected {expected} words, got {actual}")
                all_passed = False

    if all_passed:
        print("PASS: all_lines_correct")

if __name__ == "__main__":
    run_tests()

    if all_passed:
        print("PASS: all_lines_correct")
